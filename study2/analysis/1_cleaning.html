         Outlier = ifelse(Outlier_AttentionCheck == 1, "Failed Attention Checks", "Passed"),

# Compute R^2
SS_res <- sum((df$TrueScore - df$PredictedScore)^2)
SS_tot <- sum((df$TrueScore - mean(df$TrueScore))^2)
R2 <- 1 - (SS_res / SS_tot)

results |> 
  summarize(RMSE = sqrt(mean((TrueScore - value)^2)),
